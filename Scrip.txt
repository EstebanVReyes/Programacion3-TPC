CREATE DATABASE ComercioFrgp;
GO

USE ComercioFrgp;
GO


CREATE TABLE Usuarios (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(255),
    Apellido VARCHAR(255),
    Email VARCHAR(255),
    PasswordHash VARCHAR(255),
    Telefono VARCHAR(50),
    FechaRegistro DATE
);

CREATE TABLE Direcciones (
    ID INT PRIMARY KEY,
    Usuario_ID INT,
    Calle VARCHAR(255),
    Numero VARCHAR(50),
    Localidad VARCHAR(255),
    CodigoPostal VARCHAR(20),
    FOREIGN KEY (Usuario_ID) REFERENCES Usuarios(ID)
);

CREATE TABLE Categorias (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(255),
    CategoriaPadre_ID INT NULL,
    FOREIGN KEY (CategoriaPadre_ID) REFERENCES Categorias(ID)
);

CREATE TABLE Marcas (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(255)
);

CREATE TABLE Proveedor (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(255),
    Telefono VARCHAR(50),
    Descripcion TEXT
);

-- LA TABLA ARTICULOS VUELVE A TENER EL PROVEEDOR_ID
CREATE TABLE Articulos (
    ID INT PRIMARY KEY,
    CodigoSKU VARCHAR(100),
    Nombre VARCHAR(255),
    Descripcion TEXT,
    PrecioEspecial DECIMAL(10,2),
    PrecioLista DECIMAL(10,2),
    StockFisico INT,
    Categoria_ID INT,
    Marca_ID INT,
    Proveedor_ID INT,
    FOREIGN KEY (Categoria_ID) REFERENCES Categorias(ID),
    FOREIGN KEY (Marca_ID) REFERENCES Marcas(ID),
    FOREIGN KEY (Proveedor_ID) REFERENCES Proveedor(ID)
);

CREATE TABLE Metodos_Envio (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(255),
    CostoBase DECIMAL(10,2)
);

CREATE TABLE Metodos_Pago (
    ID INT PRIMARY KEY,
    Nombre VARCHAR(255),
    RecargoPorcentaje DECIMAL(5,2)
);

CREATE TABLE Pedidos (
    ID INT PRIMARY KEY,
    FechaPedido DATETIME,
    Estado VARCHAR(50),
    Total DECIMAL(10,2),
    Usuario_ID INT,
    MetodoEnvio_ID INT,
    MetodoPago_ID INT,
    FOREIGN KEY (Usuario_ID) REFERENCES Usuarios(ID),
    FOREIGN KEY (MetodoEnvio_ID) REFERENCES Metodos_Envio(ID),
    FOREIGN KEY (MetodoPago_ID) REFERENCES Metodos_Pago(ID)
);

CREATE TABLE Pedidos_Detalle (
    ID INT PRIMARY KEY,
    Pedido_ID INT,
    Articulo_ID INT,
    Cantidad INT,
    PrecioUnitario DECIMAL(10,2),
    FOREIGN KEY (Pedido_ID) REFERENCES Pedidos(ID),
    FOREIGN KEY (Articulo_ID) REFERENCES Articulos(ID)
);

CREATE TABLE Facturas (
    ID INT PRIMARY KEY,
    Pedido_ID INT,
    TipoFactura VARCHAR(10),
    CUIT_DNI VARCHAR(50),
    NombreRazonSocial VARCHAR(255),
    FechaEmision DATETIME,
    TotalFacturado DECIMAL(10,2),
    FOREIGN KEY (Pedido_ID) REFERENCES Pedidos(ID)
);
CREATE TABLE Compras (
    ID INT PRIMARY KEY,
    Proveedor_ID INT,
    FechaCompra DATETIME,
    Total DECIMAL(10,2),
    FOREIGN KEY (Proveedor_ID) REFERENCES Proveedor(ID)
);

CREATE TABLE Detalles_Compra (
    ID INT PRIMARY KEY,
    Compra_ID INT,
    Articulo_ID INT,
    Cantidad INT,
    PrecioCosto DECIMAL(10,2),
    FOREIGN KEY (Compra_ID) REFERENCES Compras(ID),
    FOREIGN KEY (Articulo_ID) REFERENCES Articulos(ID)
);
